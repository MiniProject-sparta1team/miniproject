function displayChampionInfo() { let url = 'https://ddragon.leagueoflegends.com/cdn/14.6.1/data/ko_KR/champion.json'; fetch(url) .then(res => res.json()) .then(data => { const champions = Object.entries(data.data); const sortedChampions = champions.sort((a, b) => { return a[1].name.localeCompare(b[1].name); // 챔피언 이름으로 가나다순 정렬 }); const championImagesDiv = document.getElementById('champion-images'); let championRow; sortedChampions.forEach(([championName, championData], index) => { if (index % 7 === 0) { championRow = document.createElement('div'); championRow.className = 'champion-row'; championImagesDiv.appendChild(championRow); } const championContainer = document.createElement('div'); championContainer.className = 'champion-container'; const img = document.createElement('img'); img.src = `http://ddragon.leagueoflegends.com/cdn/14.6.1/img/champion/${championName}.png`; img.alt = championName; img.className = 'champion-img'; const name = document.createElement('div'); name.className = 'champion-name'; name.textContent = championData.name; championContainer.addEventListener('click', () => { window.location.href = `result.html?name=${championName}`; }); championContainer.appendChild(img); championContainer.appendChild(name); championRow.appendChild(championContainer); if ((index + 1) % 7 === 0) { championContainer.style.marginRight = '0'; } }); }) .catch(error => console.error('데이터를 불러오는 중 오류가 발생했습니다:', error)); } window.onload = function () { displayChampionInfo(); };
